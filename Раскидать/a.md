#include <iostream>
#include <vector>
#include <queue>
#include <unordered_map>
#include <limits>

using namespace std;

typedef pair<int, int> pii; // (–≤–µ—Ä—à–∏–Ω–∞, –≤–µ—Å)
typedef pair<int, int> pfi; // (f(v), –≤–µ—Ä—à–∏–Ω–∞)

int heuristic(int v, int goal) {
    // –ü—Ä–∏–º–µ—Ä —ç–≤—Ä–∏—Å—Ç–∏—á–µ—Å–∫–æ–π —Ñ—É–Ω–∫—Ü–∏–∏ (—Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ –¥–æ —Ü–µ–ª–∏)

    if (v == 1) return 3; // B
    if (v == 2) return 1; // C
    if (v == 3) return 2; // D
    if (v == 4) return 0; // E
    return 0;
}

void aStar(const vector<vector<pii>>& graph, int start, int goal, vector<int>& path) {
    int n = graph.size();
    vector<int> g(n, numeric_limits<int>::max()); // –°—Ç–æ–∏–º–æ—Å—Ç—å –ø—É—Ç–∏ –æ—Ç start –¥–æ v
    vector<int> f(n, numeric_limits<int>::max()); // f(v) = g(v) + h(v)
    vector<int> prev(n, -1); // –ü—Ä–µ–¥—ã–¥—É—â–∞—è –≤–µ—Ä—à–∏–Ω–∞

    priority_queue<pfi, vector<pfi>, greater<pfi>> openSet; // (f(v), –≤–µ—Ä—à–∏–Ω–∞)
    g[start] = 0;
    f[start] = heuristic(start, goal);
    openSet.push({f[start], start});

    while (!openSet.empty()) {
        int u = openSet.top().second;
        openSet.pop();

        if (u == goal) {
            // –í–æ—Å—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∏–µ –ø—É—Ç–∏
            while (u != -1) {
                path.push_back(u);
                u = prev[u];
            }
            reverse(path.begin(), path.end());
            return;
        }

        for (const auto& edge : graph[u]) {
            int v = edge.first;
            int w = edge.second;

            int tempG = g[u] + w;
            if (tempG < g[v]) {
                prev[v] = u;
                g[v] = tempG;
                f[v] = g[v] + heuristic(v, goal);
                openSet.push({f[v], v});
            }
        }
    }
}

    return 0;
}
